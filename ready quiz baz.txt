#define F_CPU 16000000UL
#include <avr/io.h>
#include <util/delay.h>
#define BUZZER PC0

void display(uint16_t val)
{
uint8_t disp[]={0,0x06,0x5B,0x4F,0x66,0x6D,0x7D,0x07,0x7F};


switch (val)
{
case 0b11111110:
PORTD=~val;
break;
case 0b11111101:
PORTD=~val;
break;
case 0b11111011:
PORTD=~val;
break;
case 0b11110111:
PORTD=~val;
break;
case 0b11101111:
PORTD=~val;
break;
case 0b11011111:
PORTD=~val;
break;
case 0b10111111:
PORTD=~val;
break;
case 0b01111111:
PORTD=~val;
break;
}
while((PINB & 0b00000001)!=0)
{
_delay_ms(100);
}

PORTC=0x00;
PORTD=0X00;
PORTC=0<<0;
}
